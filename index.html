function generateQRCodeCanvas(text, size = 200) { return new Promise((resolve) => { const tempDiv = document.createElement('div'); const qr = new QRCode(tempDiv, { text: text, width: size, height: size, useSVG: false, correctLevel: QRCode.CorrectLevel.H }); // QRCode.js は img 要素を挿入するので、完了を監視 const checkImg = setInterval(() => { const img = tempDiv.querySelector('img'); if (img && img.complete) { clearInterval(checkImg); resolve(img); } }, 100); }); } document.getElementById('excelFile').addEventListener('change', async function (e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = async function (evt) { const data = new Uint8Array(evt.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheet = workbook.Sheets[workbook.SheetNames[0]]; const rows = XLSX.utils.sheet_to_json(sheet, { header: 1 }); const qrContainer = document.getElementById('qrContainer'); qrContainer.innerHTML = ''; for (const row of rows) { const number = row[0]; const line1 = row[1] || ''; const line2 = row[2] || ''; if (typeof number === 'number' && String(number).length === 7) { const canvas = document.createElement('canvas'); const size = 200; canvas.width = size; canvas.height = size + 60; const ctx = canvas.getContext('2d'); ctx.fillStyle = '#ffffff'; ctx.fillRect(0, 0, canvas.width, canvas.height); const qrImg = await generateQRCodeCanvas(String(number)); const image = new Image(); image.onload = () => { ctx.drawImage(image, 0, 0, size, size); ctx.fillStyle = '#000000'; ctx.font = '14px sans-serif'; ctx.textAlign = 'center'; ctx.fillText(line1, size / 2, size + 25); ctx.fillText(line2, size / 2, size + 45); const dataURL = canvas.toDataURL('image/png'); const downloadBtn = document.createElement('a'); downloadBtn.href = dataURL; downloadBtn.download = `QR_${number}.png`; downloadBtn.textContent = `ダウンロード (${number})`; downloadBtn.style.marginTop = '10px'; const container = document.createElement('div'); container.className = 'qr-item'; container.appendChild(canvas); container.appendChild(downloadBtn); qrContainer.appendChild(container); }; image.src = qrImg.src; } } }; reader.readAsArrayBuffer(file); });
